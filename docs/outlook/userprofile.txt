userprofile:
    path: /api/profile/
        /:
            get: [AllowAny]
                success:
                    - HTTP_200_OK {'username', 'avatar', 'exp_points', 'badges', 'created_at', ['email']}
                        exact username lookup
                failure:
                    - HTTP_400_BAD_REQUEST {'error': {'query': 'MISSING_FIELD'}}
                        query parameter `q` missing
                    - HTTP_400_BAD_REQUEST {'error': {'query': 'USER_NOT_FOUND'}}
                        profile does not exist
            post: [IsAuthenticated]
                success:
                    - HTTP_200_OK {'email', 'username', 'avatar', 'exp_points', 'badges', 'created_at', 'is_guest', 'uid'}
                        the authenticated profile was updated
                failure:
                    - HTTP_409_CONFLICT {'error': {'username': 'USERNAME_TAKEN'}}
                        username is already taken
                    - HTTP_400_BAD_REQUEST {'error': {'username': 'INVALID_USERNAME' | 'email': 'INVALID_EMAIL' | 'avatar': 'INVALID_IMAGE'}}
                        one or more submitted fields are invalid
                    - HTTP_401_UNAUTHORIZED
                        access token missing or invalid
        search/:
            get: [AllowAny]
                success:
                    - HTTP_200_OK [LightProfile]
                        exact username match search, possibly empty
                failure:
                    - HTTP_400_BAD_REQUEST {'error': {'query': 'MISSING_FIELD' | 'EMPTY_FIELD'}}
                        query parameter invalid
        guest-create/:
            post: [AllowAny]
                success:
                    - HTTP_201_CREATED LightProfile
                        a guest profile was created for the current session
                    - HTTP_200_OK LightProfile
                        an existing guest profile for the session was updated
                failure:
                    - HTTP_400_BAD_REQUEST raw serializer errors
                        validation is returned directly by DRF, not wrapped in {'error': ...}
        guest-delete/:
            post: [AllowAny]
                success:
                    - HTTP_204_NO_CONTENT
                        guest profile removed if it existed
